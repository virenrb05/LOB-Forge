Write `scripts/train_all.sh` as a complete, self-documenting bash script:

```bash
#!/usr/bin/env bash
# train_all.sh вҖ” Reproduce LOB-Forge results end-to-end
# Usage: bash scripts/train_all.sh [--skip-data] [--device cpu|mps|cuda]
# Requires: pip install -e ".[dev]"
set -euo pipefail

# --- Config ---
DEVICE=${DEVICE:-mps}   # override: DEVICE=cuda bash scripts/train_all.sh
DATA_DIR=${DATA_DIR:-data}

# --- Argument parsing ---
SKIP_DATA=false
for arg in "$@"; do
  case $arg in
    --skip-data) SKIP_DATA=true ;;
    --device=*) DEVICE="${arg#*=}" ;;
  esac
done

# --- Stage 0: Validate environment ---
echo "=== LOB-Forge Pipeline ==="
python -c "import lob_forge; print(f'lob_forge version: {lob_forge.__version__}')"
python scripts/validate_mps.py || echo "MPS validation skipped"

# --- Stage 1: Data download (Bybit) ---
if [ "$SKIP_DATA" = false ]; then
  echo "=== Stage 1: Data Ingestion ==="
  python -m lob_forge.data.bybit_downloader --config-name data
  python -m lob_forge.data.lobster_adapter --config-name data
fi

# --- Stage 2: Preprocessing ---
echo "=== Stage 2: Preprocessing ==="
python -m lob_forge.data.preprocessor --config-name data

# --- Stage 3: Train predictor ---
echo "=== Stage 3: Predictor Training ==="
python -m lob_forge.train --config-name predictor trainer=predictor device=$DEVICE

# --- Stage 4: Train generator ---
echo "=== Stage 4: Generator Training ==="
python -m lob_forge.train --config-name generator trainer=generator device=$DEVICE

# --- Stage 5: Train execution agent ---
echo "=== Stage 5: Execution Agent Training ==="
python -m lob_forge.executor.train --config-name executor device=$DEVICE

# --- Stage 6: Evaluation ---
echo "=== Stage 6: Evaluation ==="
python -c "
from lob_forge.executor import compare_to_baselines
from lob_forge.evaluation import generate_all_plots
print('Evaluation complete. Run notebooks/ for full results.')
"

echo ""
echo "=== Pipeline Complete ==="
echo "Results: outputs/plots/"
echo "Notebooks: notebooks/"
```

Make the script executable: `chmod +x scripts/train_all.sh`

The script should handle the case where data download fails due to network issues (Bybit geo-restriction known issue) by catching the exit code and continuing with a warning.
